** start of index.jsx **

export function MoodBoardItem({color, image, description}) {
  return (
    <div className="mood-board-item" style=
    {{backgroundColor: color}}>
      <img className="mood-board-image" src={image} />
      <h3 className="mood-board-text">{description}</h3>
    
    </div>
    
  )
}
export function MoodBoard() {
  const props =[
    {id: 1, color: "blue", image: "https://cdn.freecodecamp.org/curriculum/labs/pathway.jpg",description: "desc1"},
    {id: 2, color: "red", image: "https://cdn.freecodecamp.org/curriculum/labs/shore.jpg",description: "desc2"},
    {id: 3, color: "purple", image: "https://cdn.freecodecamp.org/curriculum/labs/grass.jpg",description: "desc3"},
    ];
  return (
    
    <div>
    <h1 className="mood-board-heading">Destination Mood Board</h1>
    <div className="mood-board">
    {
      props.map(prop =>< MoodBoardItem key = {prop.id}
          color = {prop.color}
          image = {prop.image}
          description = {prop.description} />)
    }

    
    </div>
    
    </div>
  )
}
