** start of script.js **

const generateElement = () => {
  return Math.floor(Math.random()*100) + 1;
};
const generateArray = () => {
  let array = [];
  for (let i=0; i < 5 ; ++i){
    array.push(generateElement())}
  return array
};
const generateContainer = () => {
  return  document.createElement("div");
};
const fillArrContainer = (htmlEl, arr1) => {
  htmlEl.innerHTML = "";
  arr1.forEach(x => {
    htmlEl.innerHTML += `<span>${x}</span>`
  })
};
const isOrdered = (int1, int2) => {
  return int1 <= int2;
}
const swapElements = (arr, index) => {
  const num1 = arr[index];
  const num2 = arr[index + 1];
  if(!isOrdered(num1,num2)){
    arr[index+1]=num1;
    arr[index] = num2
  }
  return arr
}
const highlightCurrentEls = (htmlEl, index) => {
  [index, index + 1].forEach(x => htmlEl.children[x].style.border = '1px dashed red')
};
const genBtn = document.getElementById("generate-btn");
const stArray = document.getElementById("starting-array");
const sortBtn = document.getElementById("sort-btn");
const arrCont = document.getElementById("array-container");
let genArray = [];
genBtn.addEventListener("click", () => {
  genArray = generateArray();
  fillArrContainer(stArray, genArray);
  const itemL = arrCont.children.length;
  if ( itemL > 1) {
    for (let i = itemL - 1; i > 0; --i){
      arrCont.children[i].remove();      
    }
  }
  sortBtn.hidden= false;
})
function bubbleSortAlg () {
  let chkArray = new Array(4);
  do {
    for (let i=0; i < 4; ++i){
      const divC = generateContainer();
      fillArrContainer(divC, genArray);
      highlightCurrentEls(divC, i);
      arrCont.appendChild(divC);
      chkArray[i] = isOrdered(genArray[i], genArray[i + 1])
      swapElements(genArray, i);
    }
  } while(!chkArray.every(x => x== true));
  highlightCurrentEls(stArray, 0);
  arrCont.children[1].remove();
  const finalDiv = generateContainer();
  fillArrContainer(finalDiv, genArray);    
  arrCont.appendChild(finalDiv);
  finalDiv.classList.add("final-array");  
  sortBtn.hidden= true;
}
sortBtn.addEventListener("click", () => {bubbleSortAlg()})



** end of script.js **
