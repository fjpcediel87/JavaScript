const pad = document.querySelectorAll(".drum-pad");
const display = document.getElementById("display");
function playAudio (btn) {
  const audio = btn.querySelector("audio");
  console.log(audio.getAttribute("src"))  
  audio.play();
  display.innerText = btn.id

}
Array.from(pad).forEach(x => x.addEventListener("click", ()=>playAudio(x)));
const regex = /[qweasdzxc]/i;
document.addEventListener("keydown", (x) =>{
  if (x.key.match(regex)){
    const keyAudio = document.getElementById(x.key.toUpperCase());
    keyAudio.play()
  }  
})
