const addBkBtn = document.getElementById("add-bookmark-button");
const goBkBtn = document.getElementById("close-form-button");
const addBkBtn2 = document.getElementById("add-bookmark-button-form");
const goBkBtn2 = document.getElementById("close-list-button");
const nameInput = document.getElementById("name");
const ddBtn = document.getElementById("category-dropdown");
const urlInput = document.getElementById("url");
const mainS = document.getElementById("main-section");
const formS = document.getElementById("form-section");
const bListS = document.getElementById("bookmark-list-section");
const vwCatBtn = document.getElementById("view-category-button");
const delBtn = document.getElementById("delete-bookmark-button");
const bookmarks = [{
  name: "", category: "", url: ""
}]
//functions 

const getBookmarks =  () => {
let array
try {
  array = JSON.parse(localStorage.getItem("bookmarks"));
}
catch {
  array = []
}

  if (array == null || !Array.isArray(array)) {
    return []
  } else if (array.some( x => x.name == undefined || x.category == undefined || x.url == undefined)) {
    return []
  } else {
    return array
  }
}

const displayOrCloseForm = () => {
  mainS.classList.toggle("hidden");
  formS.classList.toggle("hidden");
}
const displayOrHideCategory = () => {
  mainS.classList.toggle("hidden");
  bListS.classList.toggle("hidden");
}
const updateBookMarks = () => {
  const array = getBookmarks();
  const obj = {
    name: nameInput.value,
    category: ddBtn.value,
    url: urlInput.value
  }
  array.push(obj);
  localStorage.setItem("bookmarks", JSON.stringify(array));
  reset()
}
function reset () {
  nameInput.value = "";
  urlInput.value = "";
  displayOrCloseForm();
}

function addBkOpen () { 
  const catName = formS.querySelector(".category-name");  
  catName.innerText = ddBtn.value;
  displayOrCloseForm(); 
}
function vwCatOpen () {
const catList = document.getElementById("category-list");
const catName = bListS.querySelector(".category-name");
  catName.innerText = ddBtn.value;
  catList.innerHTML = catListFilter(ddBtn.value)
  displayOrHideCategory(); 
}
function catListFilter (cat) {
  const array = getBookmarks();
  console.log(array)
  const fil = array.filter( x => x.category == cat);
  if (fil.length == 0) {
    return '<p>No Bookmarks Found</p>'
  } else {
    let exp = ""
    fil.forEach( x => {exp += `<div class="blist"><input type="radio" id="${x.name}"  value="${x.name}" name="radioBtns"><label for="${x.name}"><a href="${x.url}">${x.name}</a></label></div>`});
    return exp
  }
}
function deletion () {  
  const cat = ddBtn.value;
  const array = getBookmarks();
  console.log(array);
  
  const listArray = document.querySelectorAll('input[name = "radioBtns"]');
  let itemToBeDel ="";
  for (const e of listArray){
    if(e.checked) {
      itemToBeDel = e.id;
    }
  }
  console.log(itemToBeDel);
  console.log(cat)
  const index = array.findIndex( x => x.name === itemToBeDel && x.category === cat)
  array.splice(index, 1)
  localStorage.setItem("bookmarks",JSON.stringify(array));
  const catList = document.getElementById("category-list");
  catList.innerHTML =
  catListFilter(cat);
}


//addEventListeners
addBkBtn.addEventListener("click", addBkOpen);
addBkBtn2.addEventListener("click", updateBookMarks);
goBkBtn.addEventListener("click", () => {displayOrCloseForm()});
goBkBtn2.addEventListener("click", () => {displayOrHideCategory()});
vwCatBtn.addEventListener("click",vwCatOpen);
delBtn.addEventListener("click", ()=> {deletion()})
