** start of script.js **

const mkdInput = document.getElementById("markdown-input");
const rawOutput = document.getElementById("html-output");
const previewBox = document.getElementById("preview");

function convertMarkdown () {
  const regexH1 = /(^|\n)#\s(.+)/g;
  const regexH2 = /(^|\n)##\s(.+)/g;
  const regexH3 = /(^|\n)###\s(.+)/g;
  const regexB = /(^|\n|)\*{2}(.+)\*{2}/g;
  const regexB2 = /(^|\n|)_{2}(.+)_{2}/g;
  const regexI = /(^|\n|)\*{1}(.+)\*{1}/g;
  const regexI2 = /(^|\n|)_{1}(.+)_{1}/g;
  const regexImg = /(^|\n)[!]{1}\[(.+)\]\((.+)\)/g;
  const regexLk =  /(^|\n)\[(.+)\]\((.+)\)/g;
  const regexQ = /(^|\n)[>]{1}\s(.+)/g;

  let str = mkdInput.value;
  console.log(str); 
  str=str
  .replace(regexH1,"<h1>$2</h1>")
  .replace(regexH2,"<h2>$2</h2>")
  .replace(regexH3,"<h3>$2</h3>")
  .replace(regexB,"<strong>$2</strong>")
  .replace(regexB2,"<strong>$2</strong>")
  .replace(regexI,"<em>$2</em>")
  .replace(regexI2,"<em>$2</em>")
  .replace(regexImg,'<img alt="$2" src="$3">')
  .replace(regexLk,'<a  href="$3">$2</a>')
  .replace(regexQ,'<blockquote>$2</blockquote>')
  return str 
}
mkdInput.addEventListener("input",()=> {
  const str = convertMarkdown();
  rawOutput.innerText = str;
 previewBox.innerHTML = str; })
